def stage_1_subgoal_constraint2():
    """constraints: <"sub-goal constraints", "the heading direction of the gripper approach of the robot", "the center of the body of the kiwi", "the heading direction of the gripper approach of the robot is perpendicular to the surface of the table and points towards the center of the body of the kiwi"> """
    pc1 = get_point_cloud("the gripper approach of the robot", -1)
    pc2 = get_point_cloud("the body of the kiwi", -1)
    pc3 = get_point_cloud("the surface of the table", -1)

    # Calculate the heading direction of the gripper approach
    covariance_matrix_gripper = np.cov(pc1.T)
    eigenvalues_gripper, eigenvectors_gripper = np.linalg.eig(covariance_matrix_gripper)
    gripper_approach = eigenvectors_gripper[:, np.argmax(eigenvalues_gripper)]
    if gripper_approach[np.argmax(np.abs(gripper_approach))] < 0:
        gripper_approach = -gripper_approach

    # Calculate the normal of the table surface
    covariance_matrix_table = np.cov(pc3.T)
    eigenvalues_table, eigenvectors_table = np.linalg.eig(covariance_matrix_table)
    table_normal = eigenvectors_table[:, np.argmin(eigenvalues_table)]
    if table_normal[np.argmax(np.abs(table_normal))] < 0:
        table_normal = -table_normal

    # Calculate the vector from gripper to kiwi center
    gripper_center = np.mean(pc1, axis=0)
    kiwi_center = np.mean(pc2, axis=0)
    gripper_to_kiwi = kiwi_center - gripper_center
    gripper_to_kiwi = gripper_to_kiwi / np.linalg.norm(gripper_to_kiwi)

    # Check if gripper approach is perpendicular to table normal and points towards kiwi center
    cost_perpendicular = abs(np.dot(gripper_approach, table_normal)) * 5.
    cost_direction = 1 - np.dot(gripper_approach, gripper_to_kiwi)
    
    return cost_perpendicular + cost_direction

def stage_1_subgoal_constraint3():
    """constraints: <"sub-goal constraints", "the binormal of the gripper of the robot", "the axis of the kiwi", "the binormal of the gripper of the robot is parallel to the axis of the kiwi"> """
    pc1 = get_point_cloud("the gripper binormal of the robot", -1)
    pc2 = get_point_cloud("the body of the kiwi", -1)

    # Calculate the binormal of the gripper
    covariance_matrix_gripper = np.cov(pc1.T)
    eigenvalues_gripper, eigenvectors_gripper = np.linalg.eig(covariance_matrix_gripper)
    gripper_binormal = eigenvectors_gripper[:, np.argmax(eigenvalues_gripper)]
    if gripper_binormal[np.argmax(np.abs(gripper_binormal))] < 0:
        gripper_binormal = -gripper_binormal

    # Calculate the axis of the kiwi
    covariance_matrix_kiwi = np.cov(pc2.T)
    eigenvalues_kiwi, eigenvectors_kiwi = np.linalg.eig(covariance_matrix_kiwi)
    kiwi_axis = eigenvectors_kiwi[:, np.argmax(eigenvalues_kiwi)]
    if kiwi_axis[np.argmax(np.abs(kiwi_axis))] < 0:
        kiwi_axis = -kiwi_axis

    # Check if gripper binormal is parallel to kiwi axis
    cost = 1 - abs(np.dot(gripper_binormal, kiwi_axis)) * 5.
    
    return cost
