def stage_2_subgoal_constraint1():
    """constraints: <"sub-goal constraints", "the center of the opening of the red cup", "the center of the body of the grey bowl", "the center of the opening of the red cup is directly above the center of the body of the grey bowl by around 10 centimeters"> """
    pc1 = get_point_cloud("the opening of the red cup", -1)
    pc2 = get_point_cloud("the body of the grey bowl", -1)

    # Compute the mean position of the opening of the red cup and the body of the grey bowl
    cup_opening_center = np.mean(pc1, axis=0)
    bowl_body_center = np.mean(pc2, axis=0)

    # Calculate the horizontal distance (x, y coordinates) between the centers
    horizontal_distance = np.linalg.norm(cup_opening_center[:2] - bowl_body_center[:2])

    # Calculate the vertical distance (z coordinate) between the centers
    vertical_distance = cup_opening_center[2] - bowl_body_center[2] - 0.1  # 10 cm above

    cost = abs(vertical_distance) + horizontal_distance
    return cost
