def stage_4_subgoal_constraint1():
    """constraints: <"sub-goal constraints", "the surface of the table", "the center of the bottom of the red cup", "the center of the bottom of the red cup is above the surface of the table by 0 cm"> """
    pc1 = get_point_cloud("the surface of the table", -1)
    pc2 = get_point_cloud("the bottom of the red cup", -1)

    table_surface_center = np.mean(pc1, axis=0)
    cup_bottom_center = np.mean(pc2, axis=0)

    vertical_distance = cup_bottom_center[2] - table_surface_center[2]
    cost = abs(vertical_distance)
    return cost
