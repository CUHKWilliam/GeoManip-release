def stage_3_path_constraint1():
    """constraints: <"path constraints", "the center of the opening of the red cup", "the center of the body of the grey bowl", "the distance between the center of the opening of the red cup and the center of the body of the grey bowl remains around 10 centimeters"> """
    pc1 = get_point_cloud("the opening of the red cup", -1)
    pc2 = get_point_cloud("the body of the grey bowl", -1)

    cup_opening_center = np.mean(pc1, axis=0)
    bowl_body_center = np.mean(pc2, axis=0)

    distance = np.linalg.norm(cup_opening_center - bowl_body_center)
    cost = abs(distance - 0.1)  # 10 cm
    return cost
